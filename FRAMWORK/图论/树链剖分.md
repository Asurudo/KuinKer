# 树链剖分

## 例题编号

无

------

对于一棵树，给出如下定义：

1. **重儿子**为子节点中子树最大的结点，如果有多个，取其一；若没有子节点，则无重儿子
2. **轻儿子**为不是重儿子的节点
3. **重边**为从轻儿子或重儿子连一条边到重儿子的边
4. **轻边**为不是重边的边
5. **重链**为若干条首尾相接的重边

![HLD](https://oi-wiki.org/graph/images/hld.png)

我们的目的就是在重链上一条链一条链操作，而非一个一个点。

首先需要预处理，$dfs1$ 预处理出了每个节点的 父节点，深度，子树大小和重儿子， $dfs2$ 预处理出了每个节点所在重链的最小深度节点，时间序以及可以通过时间序找到原节点的数组。

对于每个询问，最多经过 $O(logn)$ 条重链，每条重链上线段树的复杂度为 $O(logn)$ ，因此总的时间复杂度为 $O(nlogn+qlog^2n)$ ，不过其实重链很难到 $O(logn)$ 个，除非是完全二叉树。

模板题：

已知一个有 $n(1≤n≤10^5)$ 个节点的树，连通且无环，每个节点上包含一个数值，需要支持以下操作：

1. `1 x y z` 表示将树从 $x$ 到 $y$ 结点最短路径上所有节点的值都加上 $z$。

2. `2 x y ` 表示求树从 $x$ 到 $y$ 结点最短路径上所有节点的值之和。

3. `3 x z` 表示将以 $x$ 为根节点的子树内所有节点值都加上 $z$。

4.  `4 x`表示求以 $x$ 为根节点的子树内所有节点值之和。

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define _for(i,a,b) for(int i = (a);i < b;i ++)
#define _rep(i,a,b) for(int i = (a);i > b;i --)
#define INF 0x3f3f3f3f3f3f3f3f
#define pb push_back
#define P pair<ll,ll>
#define lowbit(x) ((x)&(-x))

#define MIKU 39
#define Design ios::sync_with_stdio(0),cin.tie(0),cout.tie(0)

//最大点数 
#define maxn 100003
//最大边数
#define maxe 400003 

ll mod;
int rt, q;

struct G
{
	//点数和边数
    int n,m;
    //存储某一条边的后一条边 
	int Next[maxe];
	//存储边的起始信息，若有x ->(w) y，则head[x]为第一条到达边的编号 
	int head[maxn];
	//存储边的到达信息 ，即 x ->(w) y中的y 
	int ver[maxe];
	//存储边的权值信息 ，即 x ->(w) y中的w
	int val[maxe];
	//有向边的总数 
	int tot;
    void clear(int x)
	{
		tot = 0;
		memset(head,0,(x+3)*sizeof(int));
	}
	void adde(int x,int y)
	{
		ver[++tot] = y,Next[tot] = head[x],head[x] = tot;
	}
} g;

//一棵平平无奇的线段树，不过a数组经过了处理
ll a[maxn];
struct SegTree
{
	struct SegNode
	{
		//l,r代表的是该节点表示的原数组下标 
		int l,r;
		//add为该区间懒惰标记 
		ll add;
		//sum,max和min分别为原数组对应区间的和，最大值以及最小值 
		ll sum,max,min;
		//x为线段树内部下标 
		#define l(x) tree[x].l
		#define r(x) tree[x].r
		#define sum(x) tree[x].sum
		#define add(x) tree[x].add
		#define MAX(x) tree[x].max
		#define MIN(x) tree[x].min
	}tree[maxn<<2];
	void build(int p,int l,int r)
	{
		add(p) = 0;
        //p为线段树内部下标 
		l(p) = l,r(p) = r;
		//到达叶子节点 
		if(l==r)
		{
			sum(p) = MAX(p) = MIN(p) = a[l];
			return ;
		}
		int mid = (l+r)/2;
		//递归构造左右节点 
		build(p*2, l, mid);
		build(p*2+1, mid+1, r);
		//左右节点维护完后开始维护父节点 
		sum(p) = sum(p*2) + sum(p*2+1); 
		MAX(p) = max(MAX(p*2),MAX(p*2+1));
		MIN(p) = min(MIN(p*2),MIN(p*2+1));
	}
	//懒惰标记下放 
	void spread(int p)
	{
		if(add(p))
		{
			//管控节点个数乘以增值为当前节点增值 
			sum(p*2) += add(p)*(r(p*2) - l(p*2) + 1);
			sum(p*2+1) += add(p)*(r(p*2+1) - l(p*2+1) + 1);
			//最值加上增值即为当前最值 
			MAX(p*2) += add(p);
			MAX(p*2+1) += add(p);
			MIN(p*2) += add(p);
			MIN(p*2+1) += add(p); 
			//更新左右节点懒惰标记 
			add(p*2) += add(p);
			add(p*2+1) += add(p);
			//当前节点懒惰标记下放完成 
			add(p) = 0;
		} 
	}
	void change(int p,int l,int r,ll d)
	{
		//要修改区间完全包住当前节点区间 
		if(l <= l(p) && r >= r(p))
		{
			sum(p) += d * (r(p)-l(p)+1);
			MAX(p) += d;
			MIN(p) += d;
			add(p) += d;
			return ;
		} 
		//否则下放当前节点懒惰标记 
		spread(p);
		int mid = (l(p)+r(p))/2;
		/*如果要修改的区间左端点比当前节点代表区间的mid小则说明涉及到了
		左区间*/ 
		if(l <= mid)
			change(p*2, l, r, d);
		//右端同理 
		if(r > mid)
			change(p*2+1, l, r, d);
		sum(p) = sum(p*2) + sum(p*2+1);
		MAX(p) = max(MAX(p*2),MAX(p*2+1));
		MIN(p) = min(MIN(p*2),MIN(p*2+1));
	} 
	ll ask4sum(int p,int l,int r)
	{
		//要修改区间完全包住当前节点区间 
		if(l <= l(p) && r >= r(p))
			return sum(p);
		spread(p);
		int mid = (l(p)+r(p))/2;
		ll val = 0;
		if(l <= mid)
			val += ask4sum(p*2, l, r);
		if(r > mid)
			val += ask4sum(p*2+1,l,r);
		return val;
	}
	ll ask4max(int p,int l,int r)
	{
		//要修改区间完全包住当前节点区间 
		if(l <= l(p) && r >= r(p))
			return MAX(p);
		spread(p);
		int mid = (l(p)+r(p))/2;
		ll val = -INF;
		//分别查询左右节点最大值 
		if(l <= mid)
			val = max(val,ask4max(p*2, l, r));
		if(r > mid)
			val = max(val,ask4max(p*2+1, l, r));
		return val;
	}
	ll ask4min(int p,int l,int r)
	{
		//要修改区间完全包住当前节点区间 
		if(l <= l(p) && r >= r(p))
			return MIN(p);
		spread(p);
		int mid = (l(p)+r(p))/2;
		ll val = INF;
		//分别查询左右节点最大值 
		if(l <= mid)
			val = min(val,ask4min(p*2, l, r));
		if(r > mid)
			val = min(val,ask4min(p*2+1, l, r));
		return val;
	}
}T;

struct TreeChainPart
{
	//节点深度根节点深度为1
	int depth[maxn];
	//父亲节点
	int fa[maxn];
	//size[x]表示以x为根的子树大小
	int size[maxn];
	//hson[x]为x的重儿子
	int hson[maxn];
	//top[x]为节点x所在重链的链顶节点
	int top[maxn];
	//重边优先遍历时的dfs序，线段树中的编号
	int dfn[maxn];
	//dfs序对应的节点编号，rnk[dfn[x]] = x
	int rank[maxn];
	//dfs序的编号
	int cnt;
	void dfs1(const G &g,int x,int fat,int dep)
	{
		hson[x] = 0;
		depth[x] = dep;
		size[x] = 1;
		fa[x] = fat;
		for(int i = g.head[x]; i; i = g.Next[i])
		{
			int y = g.ver[i];
			if(y==fat)
				continue;
			dfs1(g,y,x,dep+1);
			size[x] += size[y];
			if(size[y] > size[hson[x]])
				hson[x] = y;
		}
		return ;
	}
	//tx为当前链最顶端的节点
	void dfs2(const G &g,int x,int tx)
	{
		top[x] = tx;
		dfn[x] = ++cnt;
		rank[cnt] = x;
		if(!hson[x])
			return ;
        //先走重链
		dfs2(g,hson[x],tx);
		for(int i = g.head[x]; i; i = g.Next[i])
		{
			int y = g.ver[i];
			if(y==fa[x] || y==hson[x])
				continue;
			dfs2(g,y,y);
		}
	}
	void init(const G &g,int root)
	{
		cnt = 0;
		dfs1(g,root,-1,1);
		dfs2(g,root,root);
		//初始化线段树的数组
		ll b[maxn]; 
		_for(i,1,g.n+1)
			b[dfn[i]] = a[i];
        //处理后a[i]的i就不是树上编号了，而是dfs序的编号
		memcpy(a,b,sizeof(a));
	}
	int lca(int x,int y)
	{
		while(top[x] != top[y])
		{
			if(depth[top[x]] > depth[top[y]])
				x = fa[top[x]];
			else
				y = fa[top[y]];
		}
		return depth[x] > depth[y] ? y : x;
	}
	void change1(int x,int y,ll d)
	{
		//fx和fy分别为x和y所在重链的链顶
        int fx = top[x], fy = top[y];
        //一条一条链往上跳
		while(fx != fy)
		{
			//更新这一条链上的答案
            if(depth[fx] >= depth[fy])
				T.change(1,dfn[fx],dfn[x],d), x = fa[fx];
			else
				T.change(1,dfn[fy],dfn[y],d), y = fa[fy];
			fx = top[x], fy = top[y];
		}
        //在一条链上了，比时间序
		if(dfn[x] < dfn[y])
			T.change(1,dfn[x],dfn[y],d);
		else
			T.change(1,dfn[y],dfn[x],d);
	}
	ll ask4sum2(int x,int y)
	{
		int fx = top[x], fy = top[y];
		ll sum = 0;
		while(fx != fy)
		{
			if(depth[fx] >= depth[fy])
				sum += T.ask4sum(1,dfn[fx],dfn[x]), x = fa[fx];
			else
				sum += T.ask4sum(1,dfn[fy],dfn[y]), y = fa[fy];
			fx = top[x], fy = top[y];
		}
		if(dfn[x] < dfn[y])
			sum += T.ask4sum(1,dfn[x],dfn[y]);
		else
			sum += T.ask4sum(1,dfn[y],dfn[x]);
		return sum;
	}
	void change3(int x,ll d)
	{
		//线段树上dfn[x]到dfn[x]+size[x]-1就是线段树上以x为根节点的子树的所有节点
        T.change(1,dfn[x],dfn[x]+size[x]-1,d);
	}
	ll ask4sum4(int x)
	{
		ll sum = T.ask4sum(1,dfn[x],dfn[x]+size[x]-1);
		return sum;
	}
}TCP;
