# 线段树

## 例题编号：

**1007**，**1010**，**1020**，**1021**，**1024**，**1029**，**1034**，**1036**

------

- 线段树的每个节点都代表一个区间
- 线段树具有唯一的根节点，代表的区间是整个统计范围，如 $[1,n]$ 。
- 线段树的每个叶节点都代表一个长度为 $1$ 的元区间 $[x,x]$ 。
- 对于每个内部节点 $[l,r]$ ,它的左子节点是 $[l,mid]$ ，右子节点是 $[mid+1,r]$ ，其中 $mid = \left \lfloor \frac{l+r}{2} \right \rfloor$  。 



以上所述的区间和范围都是原数组的下标。

注意懒惰标记一定是更新过当前线段树内节点的相关值后再打上的，所以下放的时候不用考虑当前节点。

懒惰标记下放到哪儿，就说明确确实实永久更新到哪儿，不然 $sum$ 值可能只能临时改变，若不下放则再次更新时值甚至可能会减少。

注意！对于右区间比左区间小的情况模板不能很好处理，记得特判！

哪怕是所有坐标右移，也不要涉及到 $0$ 下标，否则空间应该 $<<3$ !

## 区间更新·区间查询

```c++
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define _for(i,a,b) for(register int i = (a);i < b;i ++)
#define INF 0x3f3f3f3f3f3f3f3f
#define maxn 1003
int n, m;
ll a[maxn];
struct SegTree
{
	struct SegNode
	{
		//l,r代表的是该节点表示的原数组下标 
		int l,r;
		//add为该区间懒惰标记 
		ll add;
		//sum,max和min分别为原数组对应区间的和，最大值以及最小值 
		ll sum,max,min;
		//x为线段树内部下标 
		#define l(x) tree[x].l
		#define r(x) tree[x].r
		#define sum(x) tree[x].sum
		#define add(x) tree[x].add
		#define MAX(x) tree[x].max
		#define MIN(x) tree[x].min
	}tree[maxn<<2];
	void build(int p,int l,int r)
	{
		add(p) = 0;
        //p为线段树内部下标 
		l(p) = l,r(p) = r;
		//到达叶子节点 
		if(l==r)
		{
			sum(p) = MAX(p) = MIN(p) = a[l];
			return ;
		}
		int mid = (l+r)/2;
		//递归构造左右节点 
		build(p*2, l, mid);
		build(p*2+1, mid+1, r);
		//左右节点维护完后开始维护父节点 
		sum(p) = sum(p*2) + sum(p*2+1); 
		MAX(p) = max(MAX(p*2),MAX(p*2+1));
		MIN(p) = min(MIN(p*2),MIN(p*2+1));
	}
	//懒惰标记下放 
	void spread(int p)
	{
		if(add(p))
		{
			//管控节点个数乘以增值为当前节点增值 
			sum(p*2) += add(p)*(r(p*2) - l(p*2) + 1);
			sum(p*2+1) += add(p)*(r(p*2+1) - l(p*2+1) + 1);
			//最值加上增值即为当前最值 
			MAX(p*2) += add(p);
			MAX(p*2+1) += add(p);
			MIN(p*2) += add(p);
			MIN(p*2+1) += add(p); 
			//更新左右节点懒惰标记 
			add(p*2) += add(p);
			add(p*2+1) += add(p);
			//当前节点懒惰标记下放完成 
			add(p) = 0;
		} 
	}
	void change(int p,int l,int r,ll d)
	{
		//要修改区间完全包住当前节点区间 
		if(l <= l(p) && r >= r(p))
		{
			sum(p) += d * (r(p)-l(p)+1);
			MAX(p) += d;
			MIN(p) += d;
			add(p) += d;
			return ;
		} 
		//否则下放当前节点懒惰标记 
		spread(p);
		int mid = (l(p)+r(p))/2;
		/*如果要修改的区间左端点比当前节点代表区间的mid小则说明涉及到了
		左区间*/ 
		if(l <= mid)
			change(p*2, l, r, d);
		//右端同理 
		if(r > mid)
			change(p*2+1, l, r, d);
		sum(p) = sum(p*2) + sum(p*2+1);
		MAX(p) = max(MAX(p*2),MAX(p*2+1));
		MIN(p) = min(MIN(p*2),MIN(p*2+1));
	} 
	ll ask4sum(int p,int l,int r)
	{
		//要修改区间完全包住当前节点区间 
		if(l <= l(p) && r >= r(p))
			return sum(p);
		spread(p);
		int mid = (l(p)+r(p))/2;
		ll val = 0;
		if(l <= mid)
			val += ask4sum(p*2, l, r);
		if(r > mid)
			val += ask4sum(p*2+1,l,r);
		return val;
	}
	ll ask4max(int p,int l,int r)
	{
		//要修改区间完全包住当前节点区间 
		if(l <= l(p) && r >= r(p))
			return MAX(p);
		spread(p);
		int mid = (l(p)+r(p))/2;
		ll val = -INF;
		//分别查询左右节点最大值 
		if(l <= mid)
			val = max(val,ask4max(p*2, l, r));
		if(r > mid)
			val = max(val,ask4max(p*2+1, l, r));
		return val;
	}
	ll ask4min(int p,int l,int r)
	{
		//要修改区间完全包住当前节点区间 
		if(l <= l(p) && r >= r(p))
			return MIN(p);
		spread(p);
		int mid = (l(p)+r(p))/2;
		ll val = INF;
		//分别查询左右节点最大值 
		if(l <= mid)
			val = min(val,ask4min(p*2, l, r));
		if(r > mid)
			val = min(val,ask4min(p*2+1, l, r));
		return val;
	}
    //返回第一个小于等于x的下标
	//返回-1就是没找到
	int ask4firstless(int p,int l,int r,ll x)
	{
		//到达叶子时更新
		if(l(p)==r(p))
		{
			if(MIN(p)<=x)
				return l(p);
			else
				return -1;
		}
		spread(p);
		//若当前区间已被覆盖且最小值仍太大，返回
		if(l <= l(p) && r >= r(p) && MIN(p)>x)
			return -1;
		
		int mid = (l(p)+r(p))/2;
		int jd;
		if(l <= mid)
			jd = ask4firstless(p*2,l,r,x);
		if(jd!=-1)
			return jd;
		if(r > mid)
			jd = ask4firstless(p*2+1,l,r,x);
		return jd;
	}
}T;

int main()
{
	//n为数组长度，m为询问次数 
	scanf("%d%d",&n,&m);
	_for(i,1,n+1)
		scanf("%lld",&a[i]);
	//第一个参数雷打不动是1，后面两个是a数组下标范围 
	T.build(1,1,n);
	while(m--)
	{
		// C操作为change,S操作为ask4sum,X操作为ask4max,I操作为ask4min 
		char op;
		int l, r, d;
		scanf(" %c%d%d",&op,&l,&r);
		if(op == 'C')
		{
			scanf("%d",&d);
			T.change(1, l, r, d);
		} 
		else if(op == 'S')
			printf("%lld\n",T.ask4sum(1, l, r));
		else if(op == 'X')
			printf("%lld\n",T.ask4max(1, l, r));
		else
			printf("%lld\n",T.ask4min(1, l, r));
	} 
	return 0;
}
```



## 单点更新·区间查询

```c++
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define _for(i,a,b) for(register int i = (a);i < b;i ++)
#define INF 0x3f3f3f3f3f3f3f3f
#define maxn 1003
int n, m;
ll a[maxn];
struct SegTree
{
	struct SegNode
	{
		//l,r代表的是该节点表示的原数组下标
		int l,r;
		//sum,max和min分别为原数组对应区间的和，最大值以及最小值
		ll sum,max,min;
		//x为线段树内部下标
		#define l(x) tree[x].l
		#define r(x) tree[x].r
		#define sum(x) tree[x].sum
		#define MAX(x) tree[x].max
		#define MIN(x) tree[x].min
	} tree[maxn<<2];
	void build(int p,int l,int r)
	{
        //p为线段树内部下标
		l(p) = l,r(p) = r;
		//到达叶子节点
		if(l==r)
		{
			sum(p) = MAX(p) = MIN(p) = a[l];
			return ;
		}
		int mid = (l+r)/2;
		//递归构造左右节点
		build(p*2, l, mid);
		build(p*2+1, mid+1, r);
		//左右节点维护完后开始维护父节点
		sum(p) = sum(p*2) + sum(p*2+1);
		MAX(p) = max(MAX(p*2),MAX(p*2+1));
		MIN(p) = min(MIN(p*2),MIN(p*2+1));
	}
	void change(int p,int x, ll d)
	{
		//已经到了要修改的叶子节点
		if(l(p) == r(p))
		{
			sum(p) = d;
			MAX(p) = d;
			MIN(p) = d;
			return ;
		}
		int mid = (l(p)+r(p))/2;
		/*如果要修改的端点比当前节点代表区间的mid小则说明要找的点在
		左区间*/
		if(x <= mid)
			change(p*2, x, d);
		//否则在右边
		else
			change(p*2+1, x, d);
		sum(p) = sum(p*2) + sum(p*2+1);
		MAX(p) = max(MAX(p*2),MAX(p*2+1));
		MIN(p) = min(MIN(p*2),MIN(p*2+1));
	}
	ll ask4sum(int p,int l,int r)
	{
		//要修改区间完全包住当前节点区间
		if(l <= l(p) && r >= r(p))
			return sum(p);
		int mid = (l(p)+r(p))/2;
		ll val = 0;
		if(l <= mid)
			val += ask4sum(p*2, l, r);
		if(r > mid)
			val += ask4sum(p*2+1,l,r);
		return val;
	}
	ll ask4max(int p,int l,int r)
	{
		//要修改区间完全包住当前节点区间
		if(l <= l(p) && r >= r(p))
			return MAX(p);
		int mid = (l(p)+r(p))/2;
		ll val = -INF;
		//分别查询左右节点最大值
		if(l <= mid)
			val = max(val,ask4max(p*2, l, r));
		if(r > mid)
			val = max(val,ask4max(p*2+1, l, r));
		return val;
	}
	ll ask4min(int p,int l,int r)
	{
		//要修改区间完全包住当前节点区间
		if(l <= l(p) && r >= r(p))
			return MIN(p);
		int mid = (l(p)+r(p))/2;
		ll val = INF;
		//分别查询左右节点最大值
		if(l <= mid)
			val = min(val,ask4min(p*2, l, r));
		if(r > mid)
			val = min(val,ask4min(p*2+1, l, r));
		return val;
	}
} T;

int main()
{
	//n为数组长度，m为询问次数
	scanf("%d%d",&n,&m);
	_for(i,1,n+1)
	scanf("%d",&a[i]);
	//第一个参数雷打不动是1，后面两个是a数组下标范围
	T.build(1,1,n);
	while(m--)
	{
		// C操作为change,S操作为ask4sum,X操作为ask4max,I操作为ask4min
		char op;
		int l, r;
		ll d;
		scanf(" %c",&op);
		if(op == 'C')
		{
			scanf("%d%lld",&l,&d);
			T.change(1, l, d);
		}
		else
		{
			scanf("%d%d",&l,&r);
			if(op == 'S')
				printf("%lld\n",T.ask4sum(1, l, r));
			else if(op == 'X')
				printf("%lld\n",T.ask4max(1, l, r));
			else
				printf("%lld\n",T.ask4min(1, l, r));
		}
	}
	return 0;
}
```

若出现要求 **区间更新·单点查询** 的，按照 **区间更新·区间查询** 做。
