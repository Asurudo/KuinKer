# 最小生成树

## Kruskal

### 例题编号

**NULL**

------

算法流程如下：

1. 建立并查集，每个点各自构成一个集合。
2. 把所有边按照权值从小到大排序，遍历每条边。
3. 若点 $x,y$ 连通，则忽略这条边
4. 否则连通 $x,y$ 将这条边加入最小生成树 

根据定理任意一棵最小生成树一定包含无向图中权值最小的边，则选出该边后剩下的问题就是原问题的子问题，因此最小生成树一定包含剩余边连接森林的两个不连通节点的权值最小的边（加入连通两点的连边就不是树了，只是徒增权值之和），因此该贪心算法正确。

时间复杂度为 $O(mlogm)$ 。

```c++
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
#define _for(i,a,b) for(register int i = (a);i < b;i ++)
#define _rep(i,a,b) for(register int i = (a);i > b;i --)
#define INF 0x3f3f3f3f3f3f3f3f
#define pb push_back
//最大点数
#define maxn 10003
//最大边数
#define maxe 400003
struct G
{
    //点数和边数
    int n,m;
    //存储某一条边的后一条边 
    int Next[maxe];
    //存储边的起始信息，若有x ->(w) y，则head[x]为第一条到达边的编号 
    int head[maxn];
    //存储边的到达信息 ，即 x ->(w) y中的y 
    int ver[maxe];
    //存储边的权值信息 ，即 x ->(w) y中的w
    ll val[maxe];
    //有向边的总数 
    int tot;
    void add(int x,int y,ll w)
    {
        ver[++tot] = y,Next[tot] = head[x],head[x] = tot,val[tot] = w;
    }
} g1;

struct Djs
{
    int par[maxn];
    int high[maxn];
    Djs()
    {
		_for(i,1,maxn)
        {
            par[i] = i;
            high[i] = 0;
        }
    }
    int find(int x)
    {
        //路径压缩 
        return par[x] == x ? x : par[x] = find(par[x]);
    }
    void unite(int x,int y)
    {
        x = find(x);
        y = find(y);
        if(x==y) return ;
        //按秩合并 
        if(high[x]<high[y])
            par[x] = y;
        else
        {
            par[y] = x;
            if(high[x]==high[y])
                high[x] ++;
        }
    }
    bool same(int x,int y)
    {
        return find(x) == find(y);
    }
}MSTd1;

struct Kruskal
{
	struct edge
	{
		int u;
		int v;
		ll w;
		bool operator < (edge b)
		{
			return w < b.w;
		}
	};
	//需要用到的边数组 
	vector<edge> e;
	//最小生成树权值和 
	ll MSTval;
	void kruskal(const G &g)
	{
		//将所有边加入临时数组并按照边权从小到大排序 
		_for(x,1,g.n+1)
			for(int i = g.head[x]; i; i = g.Next[i])
				e.pb({x,g.ver[i],g.val[i]});
		sort(e.begin(),e.end());
		//遍历所有边，若连边的两点未连通，则将该边加入最小生成树 
		_for(i,0,e.size())
			if(!MSTd1.same(e[i].u,e[i].v))
			{
				MSTd1.unite(e[i].u,e[i].v);
				MSTval += e[i].w;
			}
	}
}k1;

int main()
{
	scanf("%d%d",&g1.n,&g1.m);
	_for(i,1,g1.m+1)
	{
		int x,y,z;
		scanf("%d%d%lld",&x,&y,&z);
		g1.add(x,y,z);g1.add(y,x,z);
	}
	k1.kruskal(g1);
	printf("%lld",k1.MSTval);
	return 0;
}
```
