# 扫描线

## 例题编号

**1035** 

------

详情见 $LYD$    $p219$ 以及模板注释。

模板为扫描线求面积，n为矩形数量，然后是对角线坐标。  

时间复杂度 $O(nlogn)$

```c++
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define _for(i,a,b) for(int i = (a);i < b;i ++)
#define _rep(i,a,b) for(int i = (a);i > b;i --)
#define INF 0x3f3f3f3f3f3f3f3f
#define pb push_back
#define lowbit(x) ((x)&(-x))
#define mod 1000000007
#define MIKU 39
#define maxn 200039

struct disp
{
	ll a[maxn];
	int n;
	int sz;
	ll acopy[maxn];
	int lsh[maxn];
	void init(int kn)
	{
		n = kn;
		memcpy(acopy,a,sizeof(a));
		sort(acopy+1,acopy+1+n);
		sz = unique(acopy+1,acopy+1+n)-(acopy+1);
		_for(i,1,n+1)
		lsh[i] = lower_bound(acopy+1,acopy+1+sz,a[i])-acopy;
	}
	int getsmall(int x)
	{
		return lower_bound(acopy+1,acopy+1+sz,x)-acopy;
	}
	ll getbig(int i)
	{
		return acopy[i];
	}
} dsp;

struct SegTree_NOspread
{
	struct SegNode
	{
		int l, r;
		int cnt;
		ll len;
#define l(x) tree[x].l
#define r(x) tree[x].r
#define cnt(x) tree[x].cnt
#define len(x) tree[x].len
	} tree[maxn<<3];
	void build(int p,int l,int r)
	{
		l(p) = l,r(p) = r;
		cnt(p) = len(p) = 0;
		if(l==r)
			return ;
		int mid = (l+r)/2;
		build(p*2, l, mid);
		build(p*2+1, mid+1, r);
	}
	void pushup(int p)
	{
		//如果存在，len更新为这条线段的所有长度
        if(cnt(p))
			len(p) = dsp.getbig(r(p)+1)-dsp.getbig(l(p));
        //不存在就看子节点，因为这个区间没被完全覆盖，不代表子区间没有被覆盖，且子节点肯定已经更新过了
		else if(l(p) != r(p))
			len(p) = len(p*2) + len(p*2+1);
        //如果l(p)==r(p)且没有被覆盖就置0
		else
			len(p) = 0;
	}
	void change(int p,int l,int r,int d)
	{
		if(l <= l(p) && r >= r(p))
		{
			cnt(p) += d;
			pushup(p);
			return ;
		}
		int mid = (l(p)+r(p))/2;
		if(l <= mid)
			change(p*2, l, r, d);
		if(r > mid)
			change(p*2+1, l, r, d);
		pushup(p);
		return ;
	}
} ST;

struct line
{
	int x1;
	int y1;
	int y2;
	int d;
	bool operator < (line b)
	{
		//先处理出边
        if(x1!=b.x1)
        	return x1 < b.x1;
        return d < b.d;
	}
} LE[maxn];

int main()
{
	int n;
	scanf("%d",&n);
	int dspaindex = 1;
	_for(i,0,n)
	{
		int x1, y1, x2, y2;
		scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
		LE[i*2+1] = {x1,y1,y2,1};
		LE[i*2+2] = {x2,y1,y2,-1};
		dsp.a[dspaindex++] = y1;
		dsp.a[dspaindex++] = y2;
	}
    //宽按x排列
	sort(LE+1,LE+1+n*2);
    //离散化纵坐标
	dsp.init(dspaindex-1);
    //纵坐标建权值线段树
	ST.build(1,1,dsp.sz);

	ll ans = 0;
	_for(i,1,n*2+1)
	{
		//标记[LE[i].y1,LE[i].y2]这一条竖线存在/不存在了
        ST.change(1,dsp.getsmall(LE[i].y1),dsp.getsmall(LE[i].y2)-1,LE[i].d);
		if(i!=2*n)
			ans += ST.len(1)*(LE[i+1].x1-LE[i].x1);
	}
	printf("%lld\n",ans);
	return 0;
}
```
