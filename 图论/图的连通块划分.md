# 图的连通块划分

使用该算法，可以得到一个森林里的每棵树或者图上的连通块。

```c++
struct ConnectingBlock
{
	//连通块从1开始标号 
	int cnt;
	//v[i]为i属于哪个连通块 ,同时也为标记数组 
	int v[maxn];
	//c[i]为连通块i里的点数组 
	vector<vector<int>> c;
	
	void dfs(int x,const G &g)
	{
		v[x] = cnt;
		c[cnt].pb(x);
		for(int i = g.head[x]; i; i = g.Next[i])
		{
			int y = g.ver[i];
			if(v[y]) continue;
			dfs(y,g);
		}
	}
	void getCB(const G &g)
	{
		//连通块标号从1开始，所以先pb一个空的vector<int>
        vector<int> ep; 
		c.pb(ep); 
		memset(v,0,sizeof(v)); 
		_for(i,1,g.n+1)
			if(!v[i])
			{
				c.pb(ep);
				cnt ++;dfs(i,g);
			}
	} 
};
```
